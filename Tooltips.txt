Laravel

Install Via Composer
composer global require "laravel/installer"

// Install laravel
laravel new blog

// Deploy in local
php artisan serve

//Generate Application Key
php artisan key:generate

//run migration files
php artisan migrate --seed

//symlink
php artisan storage:link

//Route list
php artisan route:list



//Controller
//To create a Controller
php artisan make:controller PostController

//To create Crud Template for post controller
php artisan make:controller --resource PostController

Migrations

//To Migrate
php artisan migrate

//to create a migration file
php artisan make:migration create_post_table --create="posts"

//To Rollback
php artisan migrate:rollback

//To customise the existing table
php artisan make:migration add_column_to_post --table="posts"

// To delete all migrations
php artisan migrate:reset

// to refresh
php artisan migrate:refresh

// To know status
php artisan migrate:status

Eloquent

// To make a model
php artisan make:model
